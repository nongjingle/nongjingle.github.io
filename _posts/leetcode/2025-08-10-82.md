## 82. Remove Duplicates from Sorted List II

[problem link](https://leetcode.com/problems/remove-duplicates-from-sorted-list-ii/)

### C++
```c++
const int INF = 0x10000000;

class Solution {
 public:
  ListNode* deleteDuplicates(ListNode* head) {
    ListNode dummyUnique(INF);
    ListNode dummyRepeated(INF);
    ListNode* ptrUnique = &dummyUnique;
    ListNode* ptrRepeated = &dummyRepeated;
    ListNode* ptr = head;

    while (ptr != nullptr) {
      if (ptr->next != nullptr && ptr->val == ptr->next->val ||
          ptr->val == ptrRepeated->val) {
        ptrRepeated->next = ptr;
        ptrRepeated = ptrRepeated->next;
        ptr = ptr->next;
        ptrRepeated->next = nullptr;
      } else {
        ptrUnique->next = ptr;
        ptrUnique = ptrUnique->next;
        ptr = ptr->next;
        ptrUnique->next = nullptr;
      }
    }

    return dummyUnique.next;
  }
};
```
