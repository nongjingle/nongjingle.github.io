## 19. Remove Nth Node From End of List

[problem link](https://leetcode.com/problems/remove-nth-node-from-end-of-list/)

### C++
```c++
class Solution {
 public:
  ListNode* removeNthFromEnd(ListNode* head, int n) {
    ListNode* dummy = new ListNode(-1);
    dummy->next = head;
    ListNode* ptr = findFromEnd(dummy, n + 1);
    ptr->next = ptr->next->next;
    return dummy->next;
  }

 private:
  ListNode* findFromEnd(ListNode* dummy, int k) {
    ListNode* ptr1 = dummy;
    for (int i = 0; i < k; i++) {
      ptr1 = ptr1->next;
    }
    ListNode* ptr2 = dummy;
    while (ptr1 != nullptr) {
      ptr1 = ptr1->next;
      ptr2 = ptr2->next;
    }
    return ptr2;
  }
};
```
