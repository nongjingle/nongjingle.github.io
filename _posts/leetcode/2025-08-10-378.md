## 378. Kth Smallest Element in a Sorted Matrix

[problem link](https://leetcode.com/problems/kth-smallest-element-in-a-sorted-matrix/)

### C++
```c++
#include <queue>
#include <vector>

class Solution {
 public:
  int kthSmallest(std::vector<std::vector<int>>& matrix, int k) {
    auto compare = [](std::vector<int>& arr1, std::vector<int>& arr2) {
      return arr1[0] > arr2[0];
    };

    std::priority_queue<std::vector<int>, std::vector<std::vector<int>>,
                        decltype(compare)>
        pq(compare);

    for (int i = 0; i < matrix.size(); i++) {
      pq.push({matrix[i][0], i, 0});
    }

    int result = 0;
    int i = 0;
    int j = 0;
    while (!pq.empty() && k > 0) {
      std::vector<int> node = pq.top();
      pq.pop();
      k--;
      result = node[0];
      i = node[1];
      j = node[2];
      if (j + 1 < matrix[i].size()) pq.push({matrix[i][j + 1], i, j + 1});
    }

    return result;
  }
};
```

### Java
```java
import java.util.PriorityQueue;

class Solution {
    public int kthSmallest(int[][] matrix, int k) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((arr1, arr2) -> {
            return arr1[0] - arr2[0];
        });

        for (int i = 0; i < matrix.length; i++) {
            pq.add(new int[] { matrix[i][0], i, 0 });
        }

        int result = 0;
        int i = 0;
        int j = 0;
        while (!pq.isEmpty() && k > 0) {
            int[] node = pq.poll();
            k--;
            result = node[0];
            i = node[1];
            j = node[2];
            if (j + 1 < matrix[i].length) {
                pq.add(new int[] { matrix[i][j + 1], i, j + 1 });
            }
        }
        return result;
    }
}
```
